    // Kullanıcı Girişi
    EKRANA_YAZ "Alışveriş Sepeti Sistemi"
    DOGRULANDI = YANLIS
    IKEN (DOGRULANDI == YANLIS) DO
        EKRANA_YAZ "Kullanıcı Adı ve Şifre Giriniz:"
        KULLANICI_GIRISI = KULLANICI_GIRISI_AL()
        DOGRULANDI = KULLANICI_GIRIS_KONTROLU(KULLANICI_GIRISI) // Koşul
        EGER (DOGRULANDI == YANLIS) O ZAMAN
            EKRANA_YAZ "Hatalı Giriş. Tekrar Deneyin."
        SON
    SON

    // Ürün Gezinme ve Sepete Ekleme
    DEVAM_ET = DOGRU
    IKEN (DEVAM_ET == DOGRU) DO // Döngü
        EKRANA_YAZ "Ürün Kategorilerini Listele (1: Giyim, 2: Elektronik, ...)"
        KATEGORI_SECIMI = KULLANICI_SECIMI_AL()
        URUNLER = URUNLERI_GETIR(KATEGORI_SECIMI)
        EKRANA_YAZ "Ürünleri Listele ve Seçiniz."

        URUN_SECIMI = KULLANICI_SECIMI_AL()
        MIKTAR = MIKTAR_GIRISI_AL()

        // Stok Kontrolü
        EGER (STOK_KONTROL(URUN_SECIMI, MIKTAR) == DOGRU) O ZAMAN // Koşul
            SEPETE_EKLE(URUN_SECIMI, MIKTAR)
            EKRANA_YAZ "Ürün sepete eklendi."
        DIGER DURUMDA
            EKRANA_YAZ "Yetersiz Stok. Lütfen Miktarı Azaltın."
        SON

        EKRANA_YAZ "Alışverişe Devam Etmek İster Misiniz? (E/H)"
        CEVAP = KULLANICI_CEVABI_AL()
        EGER (CEVAP == 'H') O ZAMAN
            DEVAM_ET = YANLIS
        SON
    SON

    // Sepeti Görüntüleme ve Düzenleme
    SEPET_DUZENLE = DOGRU
    IKEN (SEPET_DUZENLE == DOGRU) DO // Döngü
        SEPETI_GORUNTULE()
        EKRANA_YAZ "Sepeti düzenlemek ister misiniz? (1: Ürün Sil, 2: Miktar Güncelle, 3: Devam Et)"
        DUZENLEME_SECIMI = KULLANICI_SECIMI_AL()

        EGER (DUZENLEME_SECIMI == 1) O ZAMAN
            URUN_SIL()
        DIGER DURUMDA EGER (DUZENLEME_SECIMI == 2) O ZAMAN
            MIKTAR_GUNCELLE()
        DIGER DURUMDA EGER (DUZENLEME_SECIMI == 3) O ZAMAN
            SEPET_DUZENLE = YANLIS
        SON
    SON

    SEPET_TOPLAMI = SEPET_TUTARI_HESAPLA()

    // Minimum Tutar Kontrolü
    EGER (SEPET_TOPLAMI < 50) O ZAMAN // Koşul
        EKRANA_YAZ "Siparişi tamamlamak için minimum sepet tutarı 50 TL olmalıdır. Lütfen ürün ekleyin."
        CIK
    SON

    // İndirim Kodu Uygulama
    INDIM_UYGULANDI = YANLIS
    EKRANA_YAZ "İndirim Kodu Uygulamak İster Misiniz? (E/H)"
    EGER (KULLANICI_CEVABI_AL() == 'E') O ZAMAN
        EKRANA_YAZ "İndirim Kodunu Giriniz:"
        KOD = KOD_GIRISI_AL()
        EGER (KOD_GECERLI_MI(KOD) == DOGRU) O ZAMAN // Koşul
            SEPET_TOPLAMI = INDIRIMI_UYGULA(SEPET_TOPLAMI, KOD)
            INDIM_UYGULANDI = DOGRU
            EKRANA_YAZ "İndirim başarıyla uygulandı."
        DIGER DURUMDA
            EKRANA_YAZ "Geçersiz İndirim Kodu."
        SON
    SON

    // Kargo Ücreti Hesaplama
    KARGO_UCRETI = 15.00
    EGER (SEPET_TOPLAMI > 200.00) O ZAMAN // Koşul
        KARGO_UCRETI = 0.00
        EKRANA_YAZ "200 TL üzeri alışverişe ücretsiz kargo kazandınız."
    SON
    TOPLAM_ODENECEK = SEPET_TOPLAMI + KARGO_UCRETI

    // Ödeme Yöntemi Seçimi
    ODEME_SECIMI_YAPILDI = YANLIS
    IKEN (ODEME_SECIMI_YAPILDI == YANLIS) DO
        EKRANA_YAZ "Ödeme Yöntemini Seçiniz (1: Kredi Kartı, 2: Havale, 3: Kapıda Ödeme):"
        ODEME_YONTEMI = KULLANICI_SECIMI_AL()
        EGER (ODEME_YONTEMI == 1 VEYA ODEME_YONTEMI == 2 VEYA ODEME_YONTEMI == 3) O ZAMAN // Koşul
            ODEME_SECIMI_YAPILDI = DOGRU
        DIGER DURUMDA
            EKRANA_YAZ "Geçersiz ödeme yöntemi."
        SON
    SON

    // Sipariş Onayı
    EKRANA_YAZ "Toplam Ödenecek Tutar: " + TOPLAM_ODENECEK + " TL"
    EKRANA_YAZ "Seçilen Ödeme Yöntemi: " + ODEME_YONTEMI
    EKRANA_YAZ "Siparişi Onaylıyor Musunuz? (E/H)"
    EGER (KULLANICI_CEVABI_AL() == 'E') O ZAMAN
        SIPARISI_OLUSTUR(SEPET, TOPLAM_ODENECEK, ODEME_YONTEMI)
        EKRANA_YAZ "Siparişiniz Başarıyla Oluşturuldu!"
    DIGER DURUMDA
        EKRANA_YAZ "Sipariş İptal Edildi."
    SON
